🧥 CloakJS

หน้านี้มีการใช้ CloakJS แล้ว โดยสามารถลองทำตามนี้ เพื่อทดสอบการใช้งานเลย

  • คลิกขวา → ไม่สามารถคลิกได้
  • กด F12 หรือ Ctrl+Shift+I → short cut ไม่สามารถใช้งานได้
  • เปิด DevTools จากเมนูเบราว์เซอร์โดยตรง → จะมีข้อความเตือนแสดงขึ้น
  • เปิด Console ดูข้อความเตือนเรื่องการหลอกลวง (self-XSS)
หมายเหตุ: ไม่มีเว็บไซต์ใดปิด DevTools ได้จริง 100% — CloakJS ทำให้เปิด DevTools ได้ยากขึ้น และแจ้งเตือนเท่านั้น

วิธีใช้งาน

<script src="cloak.js"></script>

หรือปรับแต่งข้อความที่จะแสดงเอง (ถ้าไม่กำหนด จะใช้ค่า default เดิม):

<script src="cloak.js"></script>
<script>
  CloakJS.wear({
    // --- แก้ไขข้อความที่จะแสดงได้ที่นี่ ---
    title: 'หยุด! (STOP)',                 // หัวข้อ (ทั้ง overlay และ console)
    message: [                             // เนื้อหา: ใส่เป็น array แก้ทีละบรรทัดได้
      'นี่คือฟีเจอร์ของเบราว์เซอร์ที่มีไว้สำหรับนักพัฒนาเท่านั้น',
      'หากมีใครบอกให้คุณคัดลอกและวางโค้ดที่นี่ อาจเป็นการหลอกลวง'
    ],
    // message: 'หรือใส่เป็นข้อความบรรทัดเดียวก็ได้',  // string ก็ได้

    useDebugger: true,   // ตรวจจับ DevTools แบบ undocked ด้วย (หน้าจะหน่วงเมื่อเปิด)
    redirectUrl: null,   // หรือใส่ URL เพื่อ redirect แทนการแสดง overlay
    persist: false       // true = ค้าง overlay ไว้แม้ปิด DevTools แล้ว
  });
  // อยากยกเลิก? เรียก CloakJS.takeOff();
</script>
